数组方法
标签 demo 描述
concat() arr.concat(4,5);
arr.concat(arr2);
arr.concat(arr2,arr3); 		连接两个或者更多的数组，并返回结果。
join() arr.join();
arr.join('|'); 		把数组的所有元素放入一个字符串。元素通过指定的分隔符进行分隔。
pop() arr.pop(); 删除并返回数组的最后一个元素
push() arr.push(); 向数组的末尾添加一个或更多元素，并返回新的长度。
reverse() arr.reverse(); 颠倒数组中元素的顺序。
shift() arr.shift(); 删除并返回数组的第一个元素。
slice() arr.slice(1);
arr.slice(2,4); 		从某个已有的数组返回选定的元素。
sort() arr.sort();
arr.sort(function(a,b){return a-b;}); 		对数组的元素进行排序。
splice() arr.splice(2,0,'who');
arr.splice(2,1,'who1');
arr.splice(2,3,'who2'); 		删除元素，并向数组添加新元素。
unshift() arr.unshift(); 向数组的开头添加一个或更多元素，并返回新的长度。
toString() arr.toString(); 把数组转换为字符串，并返回结果。
toSource() function demo(name,job,born){this.name = name;this.job = job;this.born = born;}
var bill = new demo('Bill','Engineer',1999);
bill.toSource(); 		返回该对象的源代码。
